create database living_figth; -- cria banco de dados
use living_figth; -- seleciona o banco de dados

-- cria as tabelas 
CREATE TABLE if not exists Usuario (
    id_user INT PRIMARY KEY AUTO_INCREMENT,
    email_user VARCHAR(100) NOT NULL UNIQUE,
    password_user VARCHAR(255) NOT NULL,
    creation_date_user DATE NOT NULL,
    status_user INT,
    tipo_user VARCHAR(50) NOT NULL -- Ex: 'aluno' ou 'instrutor'
);

CREATE TABLE if not exists Aluno (
    id_user INT PRIMARY KEY,
    name_aluno VARCHAR(150) NOT NULL,
    date_aluno DATE,
    CONSTRAINT fk_aluno_usuario FOREIGN KEY (id_user) REFERENCES Usuario (id_user)
);

CREATE TABLE if not exists Instructor (
    id_user INT PRIMARY KEY,
    name_instructor VARCHAR(150) NOT NULL,
    tel_instructor VARCHAR(20),
    CONSTRAINT fk_instructor_usuario FOREIGN KEY (id_user) REFERENCES Usuario (id_user)
);
